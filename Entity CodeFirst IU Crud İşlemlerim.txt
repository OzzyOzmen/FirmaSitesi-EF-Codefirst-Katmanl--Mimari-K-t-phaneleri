        KullaniciService kullaniciService = new KullaniciService();


        private void DataGridView1_CellClick(object sender, DataGridViewCellEventArgs e)
        {
            Temizle();
            txtkadi.Tag = (int)dataGridView1.SelectedCells[0].Value;
            txtkadi.Text = dataGridView1.SelectedCells[1].Value.ToString();
            txtparola.Text = dataGridView1.SelectedCells[2].Value.ToString();


        }

         private void Listele()
        {
            if (dataGridView1.Rows.Count > 0)
            {
                dataGridView1.DataSource = null;
            }
            dataGridView1.DataSource = kullaniciService.KullaniciListele();

        }

        private void Ekle()
        {
           KullaniciDTO kullanici = new KullaniciDTO
            {
                KullaniciAdi = txtkadi.Text,
                Parola = txtparola.Text
            };
            kullaniciService.KullaniciEkle(kullanici);
            dataGridView1.DataSource = kullaniciService.KullaniciListele();
            Temizle();

        }

         private void Guncelle()
        {
             var kullanici = kullaniciRepository.Listele().Where(r => r.KullaniciID == (int)txtkadi.Tag).FirstOrDefault();
            kullanici.KullaniciAdi = txtkadi.Text;
            kullanici.Parola = txtparola.Text;

            kullaniciService.DbContext.SaveChanges();
            dataGridView1.DataSource = kullaniciService.KullaniciListele();
            Temizle();

        }            
 
         private void Sil()
        {
             var kullanici = kullaniciRepository.Listele().Where(x => x.KullaniciID == (int)txtkadi.Tag).FirstOrDefault();
            kullanici.KullaniciAdi = txtkadi.Text;
            kullanici.Parola = txtparola.Text;

            kullaniciServis.Sil(kullanici);
            dataGridView1.DataSource = kullaniciService.KullaniciListele();
            Temizle();

        }


        private void Temizle()
        {
            txtkadi.Tag = "";
            txtkadi.Clear();
            txtparola.Clear();

        }
    }
